val enqueue : 'a t -> 'a -> 'a t

val bot_exn : 'a t -> 'a
returns the bottom (most-recently enqueued element). Raises Empty if no element is found. Complexity: O(1)

val dequeue_exn : 'a t -> 'a * 'a t

val dequeue : 'a t -> ('a * 'a t) option

val discard_exn : 'a t -> 'a t
